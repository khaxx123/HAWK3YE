        // Generate floating particles
        const particlesContainer = document.getElementById('particles');
        for (let i = 0; i < 20; i++) {
            const particle = document.createElement('div');
            particle.className = 'particle';
            particle.style.left = Math.random() * 100 + '%';
            particle.style.animationDelay = Math.random() * 10 + 's';
            particle.style.animationDuration = (10 + Math.random() * 10) + 's';
            particlesContainer.appendChild(particle);
        }

        // File upload handling
        const uploadArea = document.getElementById('uploadArea');
        const fileInput = document.getElementById('fileInput');
        const previewGrid = document.getElementById('previewGrid');
        const analyzeBtn = document.getElementById('analyzeBtn');
        const processing = document.getElementById('processing');
        const results = document.getElementById('results');
        
        let uploadedFiles = [];

        uploadArea.addEventListener('click', () => fileInput.click());

        uploadArea.addEventListener('dragover', (e) => {
            e.preventDefault();
            uploadArea.classList.add('dragover');
        });

        uploadArea.addEventListener('dragleave', () => {
            uploadArea.classList.remove('dragover');
        });

        uploadArea.addEventListener('drop', (e) => {
            e.preventDefault();
            uploadArea.classList.remove('dragover');
            handleFiles(e.dataTransfer.files);
        });

        fileInput.addEventListener('change', (e) => {
            handleFiles(e.target.files);
        });

        function handleFiles(files) {
            uploadedFiles = Array.from(files);
            displayPreviews();
            if (uploadedFiles.length > 0) {
                analyzeBtn.style.display = 'block';
            }
        }

        function displayPreviews() {
            previewGrid.innerHTML = '';
            uploadedFiles.forEach(file => {
                const reader = new FileReader();
                reader.onload = (e) => {
                    const preview = document.createElement('div');
                    preview.className = 'preview-item';
                    preview.innerHTML = `
                        <img src="${e.target.result}" alt="${file.name}">
                        <div class="filename">${file.name}</div>
                    `;
                    previewGrid.appendChild(preview);
                };
                reader.readAsDataURL(file);
            });
        }

        // Analyze button click
        analyzeBtn.addEventListener('click', async () => {
            analyzeBtn.disabled = true;
            processing.classList.add('active');
            results.classList.remove('active');

            // Check if backend is available
            const backendAvailable = await checkBackendConnection();
            
            if (backendAvailable) {
                // Create FormData and send to backend
                const formData = new FormData();
                uploadedFiles.forEach(file => {
                    formData.append('images', file);
                });

                try {
                    const response = await fetch('http://localhost:5000/analyze', {
                        method: 'POST',
                        body: formData
                    });

                    if (!response.ok) {
                        throw new Error(`HTTP error! status: ${response.status}`);
                    }

                    const data = await response.json();
                    
                    // Display results
                    setTimeout(() => {
                        processing.classList.remove('active');
                        displayResults(data);
                    }, 2000);
                } catch (error) {
                    console.error('Backend error:', error);
                    alert('Backend server error. Showing demo results instead.');
                    setTimeout(() => {
                        processing.classList.remove('active');
                        displayMockResults();
                    }, 2000);
                }
            } else {
                // Backend not available, show mock results
                console.log('Backend not available, showing mock results for demonstration');
                setTimeout(() => {
                    processing.classList.remove('active');
                    displayMockResults();
                    showBackendNotice();
                }, 2000);
            }

            analyzeBtn.disabled = false;
        });

                return false;
            }
        }

        function displayResults(data) {
            results.classList.add('active');
            
            // Update summary cards
            document.getElementById('avgWindSpeed').textContent = data.avgWindSpeed || '7.2';
            document.getElementById('siteViability').textContent = data.viability || 'HIGH POTENTIAL';
            document.getElementById('siteViability').className = 'summary-value status-high';
            document.getElementById('powerPotential').textContent = data.powerPotential || '45';
            document.getElementById('optimalHeight').textContent = data.optimalHeight || '80';
            
            // Update table
            const tbody = document.getElementById('resultsTableBody');
            tbody.innerHTML = '';
            
            data.results.forEach(result => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${result.image}</td>
                    <td>${result.hubHeight}</td>
                    <td>${result.windSpeed10m}</td>
                    <td>${result.windSpeed80m}</td>
                    <td>${result.shearAlpha}</td>
                    <td class="${result.status}">${result.suitability}</td>
                `;
                tbody.appendChild(row);
            });
            
            // Create wind speed chart
            createWindChart(data.chartData);
        }

        // Generate CSV on client side
        function generateClientSideCSV() {
            const tableRows = document.querySelectorAll('#resultsTableBody tr');
            let csvContent = "data:text/csv;charset=utf-8,";
            csvContent += "Image,Hub Height (m),Wind Speed @ 10m,Wind Speed @ 80m,Shear Alpha,Suitability\n";
            
            tableRows.forEach(row => {
                const cells = row.querySelectorAll('td');
                const rowData = Array.from(cells).map(cell => cell.textContent).join(',');
                csvContent += rowData + "\n";
            });
            
            // Add summary
            csvContent += "\nSummary\n";
            csvContent += `Average Wind Speed,${document.getElementById('avgWindSpeed').textContent} m/s\n`;
            csvContent += `Site Viability,${document.getElementById('siteViability').textContent}\n`;
            csvContent += `Power Potential,${document.getElementById('powerPotential').textContent} MW\n`;
            csvContent += `Optimal Hub Height,${document.getElementById('optimalHeight').textContent} m\n`;
            
            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", `hawk3ye_report_${new Date().toISOString().slice(0,10)}.csv`);
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }
